%% Make itemize and enumerate smaller
\newenvironment{sItemize}{
	\begin{itemize}
	\itemsep0pt
}
{
	\end{itemize}
}

\newenvironment{sEnumerate}{
	\begin{enumerate}
	\itemsep0pt
}
{
	\end{enumerate}
}


%% requirements

\newenvironment{function}[1]{
	\subsection{Product function: #1}
	\label{func:#1}
}
{}

\newenvironment{mockup}[1]{
	\section{#1}
	
	\begin{figure}[H]
		\centering
		\caption{Mockup: #1}
		\label{mock:#1}
}
{
	\end{figure}	
}

\newenvironment{requirement}[1]{
	%\stepcounter{subsubsection}
	\subsubsection{#1}
	\label{req:#1}
	
	% General used commands
	\newcommand{\header}[1]{\textbf{##1}}
	\newcommand{\cellspace}{}
	\newcommand{\Row}[2]{\hline \header{##1} & \cellspace ##2 \cellspace \\}
	
	% Requirements parts
	\newcommand{\Identifier}{
		\Row{Identifier}{
			\header{
				\thesubsubsection
			}
		}
	}

	\newcommand{\Title}{
		\Row{Title}{#1}
	}

	\newcommand{\Specification}[1]{
		\Row{Specification}{
			%\begin{singlespace*}
				##1
			%\end{singlespace*}
		}
	}

	\newcommand{\Category}[1]{
		\Row{Category}{##1}
	}

	\newcommand{\FitCriterion}[1]{
		\Row{Fit criterion}{##1}
	}

	\newcommand{\Source}[1]{
		\Row{Source}{##1}
	}

	\newcommand{\Rationale}[1]{
		\Row{Rationale}{##1}
	}

	\newcommand{\Interaction}[1]{
		\Row{Interaction}{##1}
	}

	\newcommand{\Priority}[1]{
		\Row{Priority}{##1}
	}

	\newcommand{\Stability}[1]{
		\Row{Stability}{##1}
	}

	\newcommand{\Function}[1]{
		\Row{Implemented product functions}{##1}
	}

	\newcommand{\Systemwide}{
		\Function{Systemwide}
	}

	\newcommand{\Mockup}[1]{
		\Row{See Mockup}{##1}
	}
	
	\newcommand{\Extension}[1]{
		\Row{Extends requirement}{##1}
	}

	% Define some priorities
	\newcommand{\High}{\Priority{Essential}}
	\newcommand{\Medium}{\Priority{Medium}}
	\newcommand{\Low}{\Priority{Low}}
	\newcommand{\Optional}{\Priority{Optional}}
	
	% Start with default requirements specification
	\begin{doublespace}
	\begin{tabular}{|p{0.35\textwidth}|p{0.65\textwidth}|}
		\Identifier
		\Title
}
{
		\hline 
	\end{tabular} 
	\end{doublespace}
}
